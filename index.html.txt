<script>
    // --- Gemini API 集成配置 ---
    const apiKey = ""; // 执行环境自动提供密钥
    const modelName = "gemini-2.5-flash-preview-09-2025";
    const ttsModelName = "gemini-2.5-flash-preview-tts";

    const canvas = document.getElementById('gameCanvas');

    const scoreVal = document.getElementById('scoreVal');
    const distVal = document.getElementById('distVal');
    const speedVal = document.getElementById('speedVal');
    const densityVal = document.getElementById('densityVal');
    const overlay = document.getElementById('overlay');
    const statusTitle = document.getElementById('statusTitle');
    const statusDesc = document.getElementById('statusDesc');
    const aiBox = document.getElementById('ai-box');
    const aiComment = document.getElementById('ai-comment');
    const aiTitle = document.getElementById('ai-title');
    const actionBtn = document.getElementById('actionBtn');

    const ASSETS = {
        bg: 'https://raw.githubusercontent.com/R-ZB/2026.4.19workshop/main/bg_ground.png',
        player: 'https://raw.githubusercontent.com/R-ZB/2026.4.19workshop/main/nanakusa_hakobera.png',

    };

    let images = {};
    let player, cat, obstacles = [], coins = [], bigStones = [], warnings = [];
    let state = {
        running: false,
        score: 0,
        distance: 0,
        baseSpeed: 2.2,
        worldSpeed: 2.2,
        speedMultiplier: 1.0,
        bgScrollX: 0,
        mouseX: 0,
        mouseY: 0,
        spawnChance: 0.012,
        maxSpawnChance: 0.08, 
        lastBigStoneTime: 0
    };

    // --- AI 功能函数 ---

    /**
     * 调用 Gemini 生成游戏后的评价和语音
     */
    async function processGameResults(dist, score) {
        aiBox.style.display = 'block';
        aiComment.innerText = "猫咪正在组织语言...";
        aiTitle.innerText = "";

        const systemPrompt = "你是一只调皮且充满智慧的小猫跑酷导师。你会根据玩家的距离(m)和得分(score)给出一段简短、有趣、带有鼓励性质的评价。语言风格要活泼，结尾加个'喵'。同时请根据表现返回一个有趣的称号。";
        const userQuery = `我跑了 ${dist} 米，收集了 ${score} 个金币。请给我一段日文点评和称号。返回格式必须是 JSON: {"comment": "点评文字", "title": "称号"}`;

        try {
            const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/${modelName}:generateContent?key=${apiKey}`, {
                method: "POST",
                headers: { "Content-Type": "application/json" },
                body: JSON.stringify({
                    contents: [{ parts: [{ text: userQuery }] }],
                    systemInstruction: { parts: [{ text: systemPrompt }] },
                    generationConfig: { responseMimeType: "application/json" }
                })
            });

            if (!response.ok) throw new Error("API Request Failed");
            
            const result = await response.json();
            const aiData = JSON.parse(result.candidates[0].content.parts[0].text);
            
            aiComment.innerText = aiData.comment;
            aiTitle.innerText = `[ ${aiData.title} ]`;

            // ✨ 调用语音合成
            await speakText(aiData.comment);

        } catch (error) {
            console.error(error);
            aiComment.innerText = "にゃあ... 刚才走神了，没看到你的英姿喵。";
        }
    }

    /**
     * 将文字转化为语音并播放
     */
    async function speakText(text) {
        try {
            const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/${ttsModelName}:generateContent?key=${apiKey}`, {
                method: "POST",
                headers: { "Content-Type": "application/json" },
                body: JSON.stringify({
                    contents: [{ parts: [{ text: `Say in a cheerful cat-like voice: ${text}` }] }],
                    generationConfig: {
                        responseModalities: ["AUDIO"],
                        speechConfig: {
                            voiceConfig: { prebuiltVoiceConfig: { voiceName: "Aoede" } }
                        }
                    }
                })
            });

            if (!response.ok) return;

            const result = await response.json();
            const pcmBase64 = result.candidates[0].content.parts[0].inlineData.data;
            playPCM(pcmBase64);
        } catch (e) {
            console.error("TTS Failed", e);
        }
    }

    /**
     * 播放 PCM 音频数据
     */
    function playPCM(base64) {
        const binary = atob(base64);
        const len = binary.length;
        const bytes = new Uint8Array(len);
        for (let i = 0; i < len; i++) bytes[i] = binary.charCodeAt(i);

        // 创建简单的 WAV 头部并播放
        const wavData = createWavHeader(bytes.buffer, 24000);
        const blob = new Blob([wavData], { type: 'audio/wav' });
        const url = URL.createObjectURL(blob);
        const audio = new Audio(url);
        audio.play();
    }

    function createWavHeader(buffer, sampleRate) {
        const byteLen = buffer.byteLength;
        const header = new ArrayBuffer(44);
        const view = new DataView(header);
        view.setUint32(0, 0x52494646, false); // RIFF
        view.setUint32(4, 36 + byteLen, true);
        view.setUint32(8, 0x57415645, false); // WAVE
        view.setUint32(12, 0x666d7420, false); // fmt
        view.setUint32(16, 16, true);
        view.setUint16(20, 1, true); // PCM
        view.setUint16(22, 1, true); // Mono
        view.setUint32(24, sampleRate, true);
        view.setUint32(28, sampleRate * 2, true);
        view.setUint16(32, 2, true);
        view.setUint16(34, 16, true);
        view.setUint32(36, 0x64617461, false); // data
        view.setUint32(40, byteLen, true);
        const wav = new Uint8Array(header.byteLength + buffer.byteLength);
        wav.set(new Uint8Array(header), 0);
        wav.set(new Uint8Array(buffer), 44);
        return wav;
    }

    // --- 游戏基础逻辑 ---

    function loadImages() {
        let loaded = 0;
        const keys = Object.keys(ASSETS);
        keys.forEach(key => {
            const img = new Image();
            img.src = ASSETS[key];
            img.onload = () => {
                loaded++;
                if (loaded === keys.length) console.log('Assets Ready');
            };
            images[key] = img;
        });
    }

    class Entity {
        constructor(x, y, w, h, img, speedX = 0, speedY = 0) {
            this.x = x; this.y = y; this.w = w; this.h = h;
            this.img = img; this.angle = 0;
            this.speedX = speedX; this.speedY = speedY;
        }
        draw(alpha = 1) {
            ctx.save(); ctx.globalAlpha = alpha;
            ctx.drawImage(this.img, this.x, this.y, this.w, this.h);
            ctx.restore();
        }
        drawRotating() {
            ctx.save();
            ctx.translate(this.x + this.w / 2, this.y + this.h / 2);
            ctx.rotate(this.angle);
            ctx.drawImage(this.img, -this.w / 2, -this.h / 2, this.w, this.h);
            ctx.restore();
        }
        getRect(p = 0.2) {
            return { x: this.x + this.w * p, y: this.y + this.h * p, w: this.w * (1 - p * 2), h: this.h * (1 - p * 2) };
        }
    }

    function initGame() {
        canvas.width = window.innerWidth;
        canvas.height = window.innerHeight;
        state.mouseX = canvas.width * 0.3;
        state.mouseY = canvas.height / 2;
        player = new Entity(state.mouseX, state.mouseY, 60, 60, images.player);
        cat = new Entity(state.mouseX - 150, state.mouseY, 80, 80, images.cat);
        obstacles = []; coins = []; bigStones = []; warnings = [];
        state.score = 0; state.distance = 0; state.worldSpeed = state.baseSpeed;
        state.speedMultiplier = 1.0; state.bgScrollX = 0;
        state.spawnChance = 0.012; state.lastBigStoneTime = Date.now();
        scoreVal.innerText = '0'; distVal.innerText = '0';
        speedVal.innerText = '1.0'; densityVal.innerText = '1';
        aiBox.style.display = 'none';
        for(let i=0; i<6; i++) spawnObject(true);
    }

    function spawnObject(randomX = false) {
        const isObstacle = Math.random() > 0.4;
        const scale = 0.6 + Math.random() * 1.4; 
        const w = 70 * scale, h = 90 * scale;
        const startX = randomX ? Math.random() * canvas.width : canvas.width + 150;
        const startY = Math.random() * (canvas.height - h);
        if (isObstacle) obstacles.push(new Entity(startX, startY, w, h, images.obstacle));
        else coins.push(new Entity(startX, startY, 45, 45, images.coin));
    }

    function spawnBigStone() {
        const scale = 2.0 + Math.random() * 2.0; 
        const w = 100 * scale, h = 100 * scale;
        let startX, startY, speedX, speedY;
        const dir = Math.floor(Math.random() * 3);
        const baseSpeed = state.worldSpeed * 1.8;
        if (dir === 0) { startX = canvas.width + 300; startY = Math.random() * (canvas.height - h); speedX = -baseSpeed; speedY = 0; }
        else if (dir === 1) { startX = (canvas.width * 0.5) + Math.random() * (canvas.width * 0.5 - w); startY = -h - 100; speedX = -state.worldSpeed; speedY = baseSpeed; }
        else { startX = (canvas.width * 0.5) + Math.random() * (canvas.width * 0.5 - w); startY = canvas.height + 100; speedX = -state.worldSpeed; speedY = -baseSpeed; }
        let warnX = Math.min(canvas.width - 80, Math.max(80, startX));
        let warnY = Math.min(canvas.height - 80, Math.max(80, startY));
        if (dir === 0) warnX = canvas.width - 100;
        if (dir === 1) warnY = 50;
        if (dir === 2) warnY = canvas.height - 100;
        warnings.push({ x: warnX, y: warnY, timer: 60, config: { startX, startY, w, h, speedX, speedY } });
        state.lastBigStoneTime = Date.now();
    }

    function startGame() {
        overlay.style.display = 'none';
        initGame();
        state.running = true;
        requestAnimationFrame(gameLoop);
    }

    function gameOver() {
        state.running = false;
        overlay.style.display = 'flex';
        statusTitle.innerText = '游戏结束';
        statusDesc.innerText = `最终距离: ${Math.floor(state.distance)}m | 得分: ${state.score}`;
        actionBtn.innerText = "重新出发 ✨";
        
        // ✨ 处理 AI 逻辑
        processGameResults(Math.floor(state.distance), state.score);
    }

    function update() {
        if (!state.running) return;
        const now = Date.now();
        state.distance += state.worldSpeed * 0.05;
        state.speedMultiplier = 1.0 + (state.distance / 350);
        state.worldSpeed = state.baseSpeed * state.speedMultiplier;
        distVal.innerText = Math.floor(state.distance);
        speedVal.innerText = state.speedMultiplier.toFixed(1);
        state.bgScrollX -= state.worldSpeed;
        if (state.bgScrollX <= -canvas.width) state.bgScrollX = 0;
        player.x += (state.mouseX - player.w / 2 - player.x) * 0.15;
        player.y += (state.mouseY - player.h / 2 - player.y) * 0.15;
        cat.x += ((player.x - 150) - cat.x) * 0.1;
        cat.y += (player.y - cat.y) * 0.1;

        if (now - state.lastBigStoneTime >= (8000 / state.speedMultiplier)) spawnBigStone();
        for (let i = warnings.length - 1; i >= 0; i--) {
            if (--warnings[i].timer <= 0) {
                const c = warnings[i].config;
                bigStones.push(new Entity(c.startX, c.startY, c.w, c.h, images.bigStone, c.speedX, c.speedY));
                warnings.splice(i, 1);
            }
        }
        if (Math.random() < state.spawnChance) spawnObject();

        const catRect = cat.getRect(0.25);
        for (let i = bigStones.length - 1; i >= 0; i--) {
            const s = bigStones[i]; s.x += s.speedX; s.y += s.speedY; s.angle -= 0.1;
            if (checkCollision(catRect, s.getRect(0.18))) { gameOver(); return; }
            if (s.x < -1000 || s.x > canvas.width + 1000) bigStones.splice(i, 1);
        }
        for (let i = obstacles.length - 1; i >= 0; i--) {
            obstacles[i].x -= state.worldSpeed;
            if (checkCollision(catRect, obstacles[i].getRect(0.25))) { gameOver(); return; }
            if (obstacles[i].x < -400) obstacles.splice(i, 1);
        }
        for (let i = coins.length - 1; i >= 0; i--) {
            coins[i].x -= state.worldSpeed;
            if (checkCollision(catRect, coins[i].getRect(0.1))) {
                state.score += 10; scoreVal.innerText = state.score; coins.splice(i, 1); continue;
            }
            if (coins[i].x < -200) coins.splice(i, 1);
        }
    }

    function checkCollision(r1, r2) {
        return r1.x < r2.x + r2.w && r1.x + r1.w > r2.x && r1.y < r2.y + r2.h && r1.y + r1.h > r2.y;
    }

    function draw() {
        ctx.clearRect(0, 0, canvas.width, canvas.height);
        if (images.bg.complete) {
            const bgScale = Math.max(canvas.width / images.bg.width, canvas.height / images.bg.height);
            const bgW = images.bg.width * bgScale;
            let x = state.bgScrollX % bgW;
            ctx.drawImage(images.bg, x, 0, bgW, canvas.height);
            ctx.drawImage(images.bg, x + bgW, 0, bgW, canvas.height);
        }
        coins.forEach(c => c.draw());
        obstacles.forEach(o => o.draw());
        bigStones.forEach(s => s.drawRotating());
        warnings.forEach(w => {
            ctx.save(); ctx.globalAlpha = 0.5 + Math.abs(Math.sin(w.timer * 0.15)) * 0.5;
            ctx.drawImage(images.warning, w.x - 40, w.y - 40, 80, 80); ctx.restore();
        });
        cat.draw(); player.draw();
    }

    function gameLoop() {
        if (!state.running) return;
        update();
        draw();
        requestAnimationFrame(gameLoop);
    }

    window.addEventListener('mousemove', (e) => { state.mouseX = e.clientX; state.mouseY = e.clientY; });
    window.addEventListener('touchmove', (e) => { e.preventDefault(); state.mouseX = e.touches[0].clientX; state.mouseY = e.touches[0].clientY; }, { passive: false });
    window.onload = loadImages;
    window.addEventListener('resize', () => { canvas.width = window.innerWidth; canvas.height = window.innerHeight; });
</script>
