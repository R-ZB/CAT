<script>
    const canvas = document.getElementById('gameCanvas');
    const ctx = canvas.getContext('2d');
    const scoreVal = document.getElementById('scoreVal');
    const distVal = document.getElementById('distVal');
    const speedVal = document.getElementById('speedVal');
    const densityVal = document.getElementById('densityVal');
    const overlay = document.getElementById('overlay');
    const statusTitle = document.getElementById('statusTitle');
    const statusDesc = document.getElementById('statusDesc');

    const ASSETS = {
        bg: 'https://raw.githubusercontent.com/R-ZB/2026.4.19workshop/main/bg_ground.png',
        player: 'https://raw.githubusercontent.com/R-ZB/2026.4.19workshop/main/nanakusa_hakobera.png',
        cat: 'https://raw.githubusercontent.com/R-ZB/2026.4.19workshop/main/run_cat_smile.png',
        obstacle: 'https://raw.githubusercontent.com/R-ZB/2026.4.19workshop/main/landmark_moai_pukao.png',
        coin: 'https://raw.githubusercontent.com/R-ZB/2026.4.19workshop/main/money_koban.png',
        bigStone: 'https://raw.githubusercontent.com/R-ZB/2026.4.19workshop/main/nature_stone_ishi.png',
        warning: 'https://raw.githubusercontent.com/R-ZB/2026.4.19workshop/main/pop_chuui.png'
    };

    let images = {};
    let player, cat, obstacles = [], coins = [], bigStones = [], warnings = [];
    let state = {
        running: false,
        score: 0,
        distance: 0,
        baseSpeed: 2.2,
        worldSpeed: 2.2,
        speedMultiplier: 1.0,
        bgScrollX: 0,
        mouseX: 0,
        mouseY: 0,
        spawnChance: 0.012,
        maxSpawnChance: 0.08, 
        lastBigStoneTime: 0
    };

    function loadImages() {
        let loaded = 0;
        const keys = Object.keys(ASSETS);
        keys.forEach(key => {
            const img = new Image();
            img.src = ASSETS[key];
            img.onload = () => {
                loaded++;
                if (loaded === keys.length) console.log('Assets Ready');
            };
            images[key] = img;
        });
    }

    class Entity {
        constructor(x, y, w, h, img, speedX = 0, speedY = 0) {
            this.x = x;
            this.y = y;
            this.w = w;
            this.h = h;
            this.img = img;
            this.angle = 0;
            this.speedX = speedX;
            this.speedY = speedY;
        }
        draw(alpha = 1) {
            ctx.save();
            ctx.globalAlpha = alpha;
            ctx.drawImage(this.img, this.x, this.y, this.w, this.h);
            ctx.restore();
        }
        drawRotating() {
            ctx.save();
            ctx.translate(this.x + this.w / 2, this.y + this.h / 2);
            ctx.rotate(this.angle);
            ctx.drawImage(this.img, -this.w / 2, -this.h / 2, this.w, this.h);
            ctx.restore();
        }
        getRect(p = 0.2) {
            return {
                x: this.x + this.w * p,
                y: this.y + this.h * p,
                w: this.w * (1 - p * 2),
                h: this.h * (1 - p * 2)
            };
        }
    }

    function initGame() {
        canvas.width = window.innerWidth;
        canvas.height = window.innerHeight;

        state.mouseX = canvas.width * 0.3;
        state.mouseY = canvas.height / 2;
        
        player = new Entity(state.mouseX, state.mouseY, 60, 60, images.player);
        cat = new Entity(state.mouseX - 150, state.mouseY, 80, 80, images.cat);
        
        obstacles = [];
        coins = [];
        bigStones = [];
        warnings = [];
        state.score = 0;
        state.distance = 0;
        state.worldSpeed = state.baseSpeed;
        state.speedMultiplier = 1.0;
        state.bgScrollX = 0;
        state.spawnChance = 0.012;
        state.lastBigStoneTime = Date.now();
        
        scoreVal.innerText = '0';
        distVal.innerText = '0';
        speedVal.innerText = '1.0';
        densityVal.innerText = '1';

        for(let i=0; i<6; i++) spawnObject(true);
    }

    function isOverlapping(rect, list) {
        for (let entity of list) {
            const er = entity.getRect(0);
            if (rect.x < er.x + er.w && rect.x + rect.w > er.x &&
                rect.y < er.y + er.h && rect.y + rect.h > er.y) {
                return true;
            }
        }
        return false;
    }

    function spawnObject(randomX = false) {
        const isObstacle = Math.random() > 0.4;
        const obstacleScale = 0.6 + Math.random() * 1.4; 
        const baseW = 70;
        const baseH = 90;
        const w = baseW * obstacleScale;
        const h = baseH * obstacleScale;
        
        let attempts = 0;
        while (attempts < 8) {
            const startX = randomX ? Math.random() * canvas.width : canvas.width + 150;
            const startY = Math.random() * (canvas.height - h);
            const newRect = { x: startX, y: startY, w: w, h: h };

            if (!isOverlapping(newRect, obstacles) && !isOverlapping(newRect, bigStones)) {
                if (isObstacle) {
                    obstacles.push(new Entity(startX, startY, w, h, images.obstacle));
                } else {
                    coins.push(new Entity(startX, startY, 45, 45, images.coin));
                }
                break;
            }
            attempts++;
        }
    }

    function createBigStone(config) {
        const stone = new Entity(config.startX, config.startY, config.w, config.h, images.bigStone, config.speedX, config.speedY);
        bigStones.push(stone);
    }

    function spawnBigStone() {
        const scale = 2.0 + Math.random() * 2.0; 
        const w = 100 * scale;
        const h = 100 * scale;
        
        let startX, startY, speedX, speedY;
        const dir = Math.floor(Math.random() * 3); // 0: 右, 1: 上, 2: 下
        const baseSpeed = state.worldSpeed * 1.8;

        switch(dir) {
            case 0:
                startX = canvas.width + 300;
                startY = Math.random() * (canvas.height - h);
                speedX = -baseSpeed;
                speedY = 0;
                break;
            case 1:
                startX = (canvas.width * 0.5) + Math.random() * (canvas.width * 0.5 - w);
                startY = -h - 100;
                speedX = -state.worldSpeed; 
                speedY = baseSpeed;
                break;
            case 2:
                startX = (canvas.width * 0.5) + Math.random() * (canvas.width * 0.5 - w);
                startY = canvas.height + 100;
                speedX = -state.worldSpeed;
                speedY = -baseSpeed;
                break;
        }

        // 预警位置逻辑：如果是侧边，预警在屏幕右边缘；如果是上下，预警在对应的屏幕边缘
        let warnX = Math.min(canvas.width - 80, Math.max(80, startX));
        let warnY = Math.min(canvas.height - 80, Math.max(80, startY));
        if (dir === 0) warnX = canvas.width - 100;
        if (dir === 1) warnY = 50;
        if (dir === 2) warnY = canvas.height - 100;

        const warning = {
            x: warnX,
            y: warnY,
            timer: 60, // 约1秒 (60帧)
            config: { startX, startY, w, h, speedX, speedY }
        };
        warnings.push(warning);
        state.lastBigStoneTime = Date.now();
    }

    function checkCollision(r1, r2) {
        return r1.x < r2.x + r2.w && r1.x + r1.w > r2.x && r1.y < r2.y + r2.h && r1.y + r1.h > r2.y;
    }

    window.addEventListener('mousemove', (e) => {
        state.mouseX = e.clientX;
        state.mouseY = e.clientY;
    });

    window.addEventListener('touchmove', (e) => {
        e.preventDefault();
        state.mouseX = e.touches[0].clientX;
        state.mouseY = e.touches[0].clientY;
    }, { passive: false });

    function startGame() {
        overlay.style.display = 'none';
        initGame();
        state.running = true;
        requestAnimationFrame(gameLoop);
    }

    function gameOver() {
        state.running = false;
        overlay.style.display = 'flex';
        statusTitle.innerText = 'ゲームオーバー';
        statusDesc.innerText = `巨石を避けられなかった！\n距離: ${Math.floor(state.distance)}m\nスコア: ${state.score}`;
        const btn = document.querySelector('.btn');
        btn.innerText = 'もう一度挑戦';
    }

    function update() {
        const now = Date.now();
        
        state.distance += state.worldSpeed * 0.05;
        state.speedMultiplier = 1.0 + (state.distance / 350);
        state.worldSpeed = state.baseSpeed * state.speedMultiplier;
        
        state.spawnChance = Math.min(state.maxSpawnChance, 0.012 + (state.distance / 100) * 0.005);
        
        distVal.innerText = Math.floor(state.distance);
        speedVal.innerText = state.speedMultiplier.toFixed(1);
        densityVal.innerText = (state.spawnChance / 0.012).toFixed(1);

        state.bgScrollX -= state.worldSpeed;
        if (state.bgScrollX <= -canvas.width) state.bgScrollX = 0;

        player.x += (state.mouseX - player.w / 2 - player.x) * 0.15;
        player.y += (state.mouseY - player.h / 2 - player.y) * 0.15;

        const dx = (player.x - 150) - cat.x;
        const dy = player.y - cat.y;
        cat.x += dx * 0.1;
        cat.y += dy * 0.1;

        // 巨石生成逻辑
        if (now - state.lastBigStoneTime >= (8000 / state.speedMultiplier)) { 
            spawnBigStone();
        }

        // 处理警告倒计时
        for (let i = warnings.length - 1; i >= 0; i--) {
            warnings[i].timer--;
            if (warnings[i].timer <= 0) {
                createBigStone(warnings[i].config);
                warnings.splice(i, 1);
            }
        }

        if (Math.random() < state.spawnChance) {
            spawnObject();
        }

        const catRect = cat.getRect(0.25);

        for (let i = bigStones.length - 1; i >= 0; i--) {
            const s = bigStones[i];
            s.x += s.speedX;
            s.y += s.speedY;
            s.angle -= 0.1;
            
            if (checkCollision(catRect, s.getRect(0.18))) {
                gameOver();
                return;
            }
            if (s.x < -1000 || s.x > canvas.width + 1000 || s.y < -1000 || s.y > canvas.height + 1000) {
                bigStones.splice(i, 1);
            }
        }

        for (let i = obstacles.length - 1; i >= 0; i--) {
            const o = obstacles[i];
            o.x -= state.worldSpeed;
            if (checkCollision(catRect, o.getRect(0.25))) {
                gameOver();
                return;
            }
            if (o.x + o.w < -400) obstacles.splice(i, 1);
        }

        for (let i = coins.length - 1; i >= 0; i--) {
            const c = coins[i];
            c.x -= state.worldSpeed;
            if (checkCollision(catRect, c.getRect(0.1))) {
                state.score += 10;
                scoreVal.innerText = state.score;
                coins.splice(i, 1);
                continue;
            }
            if (c.x + c.w < -200) coins.splice(i, 1);
        }
    }

    function draw() {
        ctx.clearRect(0, 0, canvas.width, canvas.height);

        // 背景绘制
        if (images.bg.complete) {
            const bgScale = Math.max(canvas.width / images.bg.width, canvas.height / images.bg.height);
            const bgWidth = images.bg.width * bgScale;
            const bgHeight = images.bg.height * bgScale;
            let x = state.bgScrollX % bgWidth;
            ctx.drawImage(images.bg, x, 0, bgWidth, bgHeight);
            ctx.drawImage(images.bg, x + bgWidth, 0, bgWidth, bgHeight);
            if (x + bgWidth < canvas.width) ctx.drawImage(images.bg, x + bgWidth * 2, 0, bgWidth, bgHeight);
        }

        coins.forEach(c => c.draw());
        obstacles.forEach(o => o.draw());
        bigStones.forEach(s => s.drawRotating());

        // 绘制预警图标
        warnings.forEach(w => {
            // 简单的闪烁效果
            const alpha = 0.5 + Math.abs(Math.sin(w.timer * 0.15)) * 0.5;
            ctx.save();
            ctx.globalAlpha = alpha;
            // 绘制警告图标
            ctx.drawImage(images.warning, w.x - 40, w.y - 40, 80, 80);
            ctx.restore();
        });

        if (state.running) {
            ctx.beginPath();
            ctx.strokeStyle = 'rgba(255, 255, 255, 0.15)';
            ctx.setLineDash([6, 6]);
            ctx.moveTo(player.x + player.w/2, player.y + player.h/2);
            ctx.lineTo(cat.x + cat.w/2, cat.y + cat.h/2);
            ctx.stroke();
            ctx.setLineDash([]);
        }

        cat.draw();
        player.draw();
    }

    function gameLoop() {
        if (!state.running) return;
        update();
        draw();
        requestAnimationFrame(gameLoop);
    }

    window.onload = loadImages;
    window.addEventListener('resize', () => {
        canvas.width = window.innerWidth;
        canvas.height = window.innerHeight;
    });
</script>
